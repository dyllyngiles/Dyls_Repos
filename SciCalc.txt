import java.util.Scanner;

public class SciCalculator

            //labeling each operator, and keeping track of them within the loop.

                switch(calcSel)

                        reSponse = firstOp + secondOp;

                        System.out.println("2. Subtraction");
                        System.out.println("3. Multiplication");
                        System.out.println("4. Division");
                        System.out.println("5. Exponentiation");
                        System.out.println("6. Logarithm");
                        System.out.println("7. Display Average");
                        System.out.println("\nEnter Menu Selection:");
                        calcSel = myScanner.nextInt();

                        if (calcSel == 0)
                        {
                            System.out.println("Thanks for using this calculator. Goodbye!");
                            calcBool = false;
                            System.exit(0);
                        }
                        break;

                    case 2:
                        reSponse = firstOp - secondOp;
                        sum += reSponse;
                        numCalc = numCalc + 1;
                        System.out.println("Current Result:" + reSponse + "");
                        System.out.println("\nCalculator Menu");
                        System.out.println("---------------");
                        System.out.println("0. Exit Program");
                        System.out.println("1. Addition");
                        System.out.println("2. Subtraction");
                        System.out.println("3. Multiplication");
                        System.out.println("4. Division");
                        System.out.println("5. Exponentiation");
                        System.out.println("6. Logarithm");
                        System.out.println("7. Display Average");
                        System.out.println("\nEnter Menu Selection:");
                        calcSel = myScanner.nextInt();

                        if (calcSel == 0)
                        {
                            System.out.println("Thanks for using this calculator. Goodbye!");
                            calcBool = false;
                            System.exit(0);
                        }

                        break;

                    case 3:
                        reSponse = firstOp * secondOp;
                        sum += reSponse;
                        numCalc = numCalc + 1;
                        System.out.println("Current Result:" + reSponse + "");
                        System.out.println("\nCalculator Menu");
                        System.out.println("---------------");
                        System.out.println("0. Exit Program");
                        System.out.println("1. Addition");
                        System.out.println("2. Subtraction");
                        System.out.println("3. Multiplication");
                        System.out.println("4. Division");
                        System.out.println("5. Exponentiation");
                        System.out.println("6. Logarithm");
                        System.out.println("7. Display Average");

                        calcSel = myScanner.nextInt();

                        if (calcSel == 0)
                        {
                            System.out.println("Thanks for using this calculator. Goodbye!");
                            calcBool = false;
                            System.exit(0);
                        }

                        break;

                    case 4:
                        reSponse = firstOp / secondOp;
                        sum += reSponse;
                        numCalc = numCalc + 1;
                        System.out.println("Current Result:" + reSponse + "");
                        System.out.println("\nCalculator Menu");
                        System.out.println("---------------");

                        System.out.println("6. Logarithm");
                        System.out.println("7. Display Average");

                        if (calcSel == 0)
                        {
                            System.out.println("Thanks for using this calculator. Goodbye!");
                            calcBool = false;
                            System.exit(0);
                        }

                        break;

                    case 5:
                        reSponse = Math.pow(firstOp, secondOp);

                        numCalc = numCalc + 1;
                        System.out.println("Current Result:" + reSponse + "");
                        System.out.println("\nCalculator Menu");
                        System.out.println("---------------");
                        System.out.println("0. Exit Program");
                        System.out.println("1. Addition");
                        System.out.println("2. Subtraction");
                        System.out.println("3. Multiplication");
                        System.out.println("4. Division");
                        System.out.println("5. Exponentiation");
                        System.out.println("6. Logarithm");
                        System.out.println("7. Display Average");

                            System.out.println("Thanks for using this calculator. Goodbye!");

                        System.out.println("2. Subtraction");
                        System.out.println("3. Multiplication");

                        System.out.println("5. Exponentiation");
                        System.out.println("6. Logarithm");
                        System.out.println("7. Display Average");
